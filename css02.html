input type 스타일
/* 체크된 체크박스/라디오 */
input:checked {
  outline: 2px solid green;
}

/* 비활성화된 요소 */
input:disabled {
  background-color: #e0e0e0;
  cursor: not-allowed;
}

/* 활성화된 요소 */
input:enabled {
  border-color: #333;
}

/* 필수 입력 필드 */
input:required {
  border-left: 3px solid red;
}

/* 선택 사항 필드 */
input:optional {
  border-left: 3px solid gray;
}

/* 유효한 입력 */
input:valid {
  border-color: green;
}

/* 유효하지 않은 입력 */
input:invalid {
  border-color: red;
}
